TASK 01 Lexical Analyser for a given language
📌 Concept A Lexical Analyser is the first phase of a compiler. It reads source code character by character and groups them into tokens (keywords, identifiers, operators, numbers, etc.). It ignores whitespace, tabs, and comments. Keywords are matched against a fixed list (32 C keywords here). Identifiers followed by ( are treated as function/method names. 
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>

int main() {
    char *s = (char *)calloc(sizeof(char), 20000);
    char *t = (char *)calloc(sizeof(char), 20000);
    char key[32][12] = {
        "auto","break","case","char","const","continue",
        "default","do","double","else","enum","extern",
        "float","for","goto","if","int","long",
        "register","return","short","signed","sizeof",
        "static","struct","switch","typedef","union",
        "unsigned","void","volatile","while"
    };
    int i = 0, j, k, c;
    char f[50], ch;
    FILE *fp;

    printf("Enter filename: ");
    scanf("%s", f);
    fp = fopen(f, "r");
    if (fp == NULL) { printf("Cannot open file.\n"); return 1; }

    k = 0;
    while ((ch = fgetc(fp)) != EOF) { s[k++] = ch; }
    s[k] = '\0';
    printf("%s\n---\n", s);
    i = 0;

    while (s[i] != '\0') {
        j = 0; c = 0;

        if (s[i] == '#') {
            i++;
            while (s[i] != '\n' && s[i] != '\0') { printf("%c", s[i++]); }
            printf(" is a preprocessor directive\n");
            i++;
        }
        else if (isalpha(s[i]) || s[i] == '_') {
            while (isalpha(s[i]) || isdigit(s[i]) || s[i] == '_') { t[j++] = s[i++]; }
            t[j] = '\0';
            for (k = 0; k < 32; k++) if (strcmp(t, key[k]) == 0) { c = 1; break; }
            if (c == 1) printf("%s is a keyword\n", t);
            else if (s[i] == '(') printf("%s is a method\n", t);
            else printf("%s is an identifier\n", t);
        }
        else if (s[i] == '"') {
            i++;
            printf("\"");
            while (s[i] != '"' && s[i] != '\0') { printf("%c", s[i++]); }
            printf("\" is an argument\n");
            i++;
        }
        else if (s[i] == '/' && s[i+1] == '/') {
            i += 2;
            while (s[i] != '\n' && s[i] != '\0') { printf("%c", s[i++]); }
            printf(" is a comment\n");
        }
        else if (s[i] == '/' && s[i+1] == '*') {
            i += 2;
            while (!(s[i] == '*' && s[i+1] == '/')) { printf("%c", s[i++]); }
            i += 2;
            printf(" is a block comment\n");
        }
        else if (s[i]=='[' || s[i]==']' || s[i]=='(' || s[i]==')' || s[i]=='{' || s[i]=='}') {
            printf("%c is a special symbol\n", s[i++]);
        }
        else if (isdigit(s[i])) {
            while (isdigit(s[i]) || s[i] == '.') { printf("%c", s[i++]); }
            printf(" is a number\n");
        }
        else if (s[i]=='=' || s[i]=='+' || s[i]=='-' || s[i]=='*' ||
                 s[i]=='/' || s[i]=='%' || s[i]=='<' || s[i]=='>' ||
                 s[i]=='&' || s[i]=='|') {
            printf("%c is an operator\n", s[i++]);
        }
        else if (s[i] == ';') { printf("; is a terminator\n"); i++; }
        else i++;
    }
    fclose(fp);
    free(s); free(t);
    return 0;
}
TASK 02 Recognize strings under a*, a*b+, abb
📌 Concept This uses a Finite Automaton (DFA) with 7 states (0–6) to recognize three regular expressions:
• a* → zero or more 'a's (state 1)
• a*b+ → zero or more 'a's followed by one or more 'b's (states 2, 4)
• abb → exactly the string "abb" (state 5)
State 6 = reject/error state. 
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main() {
    char s[20];
    int state = 0, i = 0;
    char c;

    printf("Enter a string: ");
    scanf("%s", s);

    while (s[i] != '\0') {
        switch (state) {
            case 0: c = s[i++];
                if(c=='a') state=1;
                else if(c=='b') state=2;
                else state=6;
                break;
            case 1: c = s[i++];
                if(c=='a') state=3;
                else if(c=='b') state=4;
                else state=6;
                break;
            case 2: c = s[i++];
                if(c=='b') state=2;
                else state=6;
                break;
            case 3: c = s[i++];
                if(c=='a') state=3;
                else if(c=='b') state=2;
                else state=6;
                break;
            case 4: c = s[i++];
                if(c=='b') state=5;
                else state=6;
                break;
            case 5: c = s[i++];
                if(c=='b') state=2;
                else state=6;
                break;
            case 6:
                printf("\n%s is NOT recognised.\n", s);
                return 0;
        }
    }
    if (state == 1)
        printf("\n%s is accepted under rule 'a*'\n", s);
    else if (state == 2 || state == 4)
        printf("\n%s is accepted under rule 'a*b+'\n", s);
    else if (state == 5)
        printf("\n%s is accepted under rule 'abb'\n", s);
    else
        printf("\n%s is NOT recognised.\n", s);

    return 0;
}
TASK 03 Symbol Table Formation
📌 Concept A Symbol Table stores information about each token found in the source program — its name, memory address, and type (identifier or operator). It is used in all phases of the compiler for type checking and code generation. Here, malloc() is used to assign a unique memory address to each symbol. 
#include <stdio.h>
#include <ctype.h>
#include <stdlib.h>
#include <string.h>

int main() {
    int i = 0, j = 0, x = 0, n;
    void *p, *addr[20];
    char ch, b[30], d[20], c;

    printf("Enter expression terminated by $: ");
    while ((c = getchar()) != '$') {
        b[i++] = c;
    }
    n = i - 1;

    printf("Given Expression: ");
    for (i = 0; i <= n; i++) printf("%c", b[i]);

    printf("\n\nSymbol Table\n");
    printf("Symbol\tAddress\t\tType\n");
    printf("------\t-------\t\t----\n");

    j = 0;
    while (j <= n) {
        c = b[j];
        if (isalpha(c)) {
            p = malloc(1);
            addr[x] = p;
            d[x] = c;
            printf("%c\t%p\t\tidentifier\n", c, p);
            x++; j++;
        } else {
            ch = c;
            if (ch == '+' || ch == '-' || ch == '*' || ch == '=') {
                p = malloc(1);
                addr[x] = p;
                d[x] = ch;
                printf("%c\t%p\t\toperator\n", ch, p);
                x++; j++;
            } else {
                j++;
            }
        }
    }
    return 0;
}
TASK 04 Predictive Parser Table (LL(1))
📌 Concept A Predictive Parser is a top-down parser that uses FIRST and FOLLOW sets to build a parsing table (LL(1) table). For each non-terminal and terminal, the table tells which production to apply. 'e' is used for epsilon (empty string). Productions are entered in the form A=xyz. 
#include <stdio.h>
#include <string.h>
#include <ctype.h>

int n, m = 0;
char a[10][20], f[20];
int i = 0;

void first(char c);
void follow(char c);

void follow(char c) {
    int k, l;
    if (a[0][0] == c) f[m++] = '$';
    for (k = 0; k < n; k++) {
        for (l = 2; l < (int)strlen(a[k]); l++) {
            if (a[k][l] == c) {
                if (a[k][l+1] != '\0') first(a[k][l+1]);
                if (a[k][l+1] == '\0' && c != a[k][0]) follow(a[k][0]);
            }
        }
    }
}

void first(char c) {
    int k;
    if (!isupper(c)) { f[m++] = c; return; }
    for (k = 0; k < n; k++) {
        if (a[k][0] == c) {
            if (a[k][2] == 'e') follow(a[k][0]);
            else if (islower(a[k][2])) f[m++] = a[k][2];
            else first(a[k][2]);
        }
    }
}

int main() {
    int n1, x, z;
    char t[20], c;

    printf("Enter number of terminals: ");
    scanf("%d", &n1);
    printf("Enter terminals (e.g. $abc): ");
    scanf("%s", t);

    printf("Enter number of productions: ");
    scanf("%d", &n);
    printf("Enter productions (epsilon=e, format A=xyz):\n");
    for (i = 0; i < n; i++) scanf("%s", a[i]);

    for (i = 0; i < n1; i++) printf("\t%c", t[i]);
    printf("\n");

    for (i = 0; i < n; i++) {
        if (islower(a[i][2]) && a[i][2] != 'e') {
            x = 0;
            printf("%c\t", a[i][0]);
            while (a[i][2] != t[x]) { printf("\t"); x++; }
            printf("%c->%s\n", a[i][0], a[i]+2);
        }
        if (isupper(a[i][2])) {
            m = 0; first(a[i][2]);
            printf("%c\t", a[i][0]);
            for (z = 0; z < m; z++) {
                x = 0;
                while (f[z] != t[x]) { printf("\t"); x++; }
                printf("%c->%s\t", a[i][0], a[i]+2);
            }
            printf("\n");
        }
        if (a[i][2] == 'e') {
            m = 0; follow(a[i][0]);
            printf("%c\t", a[i][0]);
            for (z = 0; z < m; z++) {
                x = 0;
                while (f[z] != t[x]) { printf("\t"); x++; }
                printf("%c->e\t", a[i][0]);
            }
            printf("\n");
        }
    }
    return 0;
}
TASK 05 Compute FIRST() and FOLLOW() for a Grammar
📌 Concept FIRST(A) = set of terminals that begin any string derived from A. If A →* ε, add ε too.
FOLLOW(A) = set of terminals that can appear immediately to the right of A in any sentential form. Always add $ to FOLLOW of the start symbol.
These sets are essential for building LL(1) and LALR parse tables. '#' represents ε (epsilon) in this program. 
#include <stdio.h>
#include <string.h>
#include <ctype.h>

void followfirst(char, int, int);
void follow(char c);
void findfirst(char, int, int);

int count, n = 0;
char calc_first[10][100];
char calc_follow[10][100];
int m = 0;
char production[10][10];
char f[10], first_arr[10];
int k;
char ck;
int e;

int main() {
    int jm = 0, km = 0, i, kay;
    char c, done[10], donee[10];
    int ptr = -1;

    count = 8;
    strcpy(production[0], "X=TnS");
    strcpy(production[1], "X=Rm");
    strcpy(production[2], "T=q");
    strcpy(production[3], "T=#");
    strcpy(production[4], "S=p");
    strcpy(production[5], "S=#");
    strcpy(production[6], "R=om");
    strcpy(production[7], "R=ST");

    for (k = 0; k < count; k++)
        for (kay = 0; kay < 100; kay++)
            calc_first[k][kay] = '!';

    int point1 = 0, point2, xxx;
    for (k = 0; k < count; k++) {
        c = production[k][0]; point2 = 0; xxx = 0;
        for (kay = 0; kay <= ptr; kay++) if (c == done[kay]) xxx = 1;
        if (xxx == 1) continue;
        findfirst(c, 0, 0);
        ptr++; done[ptr] = c;
        printf("First(%c) = { ", c);
        calc_first[point1][point2++] = c;
        for (i = jm; i < n; i++) {
            int lark = 0, chk = 0;
            for (lark = 0; lark < point2; lark++)
                if (first_arr[i] == calc_first[point1][lark]) { chk = 1; break; }
            if (!chk) { printf("%c, ", first_arr[i]); calc_first[point1][point2++] = first_arr[i]; }
        }
        printf("}\n"); jm = n; point1++;
    }

    printf("\n---\n\n");
    ptr = -1; point1 = 0;
    for (k = 0; k < count; k++)
        for (kay = 0; kay < 100; kay++)
            calc_follow[k][kay] = '!';

    for (e = 0; e < count; e++) {
        ck = production[e][0]; point2 = 0; xxx = 0;
        for (kay = 0; kay <= ptr; kay++) if (ck == donee[kay]) xxx = 1;
        if (xxx == 1) continue;
        follow(ck); ptr++; donee[ptr] = ck;
        printf("Follow(%c) = { ", ck);
        calc_follow[point1][point2++] = ck;
        for (i = km; i < m; i++) {
            int lark = 0, chk = 0;
            for (lark = 0; lark < point2; lark++)
                if (f[i] == calc_follow[point1][lark]) { chk = 1; break; }
            if (!chk) { printf("%c, ", f[i]); calc_follow[point1][point2++] = f[i]; }
        }
        printf("}\n\n"); km = m; point1++;
    }
    return 0;
}

void follow(char c) {
    int i, j;
    if (production[0][0] == c) f[m++] = '$';
    for (i = 0; i < count; i++) {
        for (j = 2; j < 10; j++) {
            if (production[i][j] == c) {
                if (production[i][j+1] != '\0') followfirst(production[i][j+1], i, j+2);
                if (production[i][j+1] == '\0' && c != production[i][0]) follow(production[i][0]);
            }
        }
    }
}

void findfirst(char c, int q1, int q2) {
    int j;
    if (!isupper(c)) { first_arr[n++] = c; return; }
    for (j = 0; j < count; j++) {
        if (production[j][0] == c) {
            if (production[j][2] == '#') {
                if (production[q1][q2] == '\0') first_arr[n++] = '#';
                else if (q1 != 0 || q2 != 0) findfirst(production[q1][q2], q1, q2+1);
                else first_arr[n++] = '#';
            } else if (!isupper(production[j][2])) {
                first_arr[n++] = production[j][2];
            } else {
                findfirst(production[j][2], j, 3);
            }
        }
    }
}

void followfirst(char c, int c1, int c2) {
    int i, j;
    if (!isupper(c)) { f[m++] = c; return; }
    i = 0; j = 1;
    for (i = 0; i < count; i++) if (calc_first[i][0] == c) break;
    while (calc_first[i][j] != '!') {
        if (calc_first[i][j] != '#') {
            f[m++] = calc_first[i][j];
        } else {
            if (production[c1][c2] == '\0') follow(production[c1][0]);
            else followfirst(production[c1][c2], c1, c2+1);
        }
        j++;
    }
}
TASK 06 Operator Precedence Parser (Grammar Check)
📌 Concept An Operator Grammar is one where no production has two adjacent non-terminals and none has ε. An Operator Precedence Parser works bottom-up using precedence relations (<·, ·=·, ·>) between operators. This program checks whether a given grammar qualifies as an operator grammar by verifying that all productions contain arithmetic operators (+, -, *, /). 
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

void notOperator() {
    printf("Not an operator grammar\n");
    exit(0);
}

int main() {
    char grm[20][20];
    int i, n, flag = 0;

    printf("Enter number of productions: ");
    scanf("%d", &n);
    printf("Enter productions (format A=BxC):\n");
    for (i = 0; i < n; i++) scanf("%s", grm[i]);

    for (i = 0; i < n; i++) {
        char op = grm[i][3];
        if (grm[i][2] == '$') notOperator();
        if (op == '+' || op == '-' || op == '*' || op == '/')
            flag = 1;
        else
            notOperator();
    }
    if (flag == 1) printf("Operator grammar\n");
    return 0;
}

TASK 07 Shift-Reduce Parser
📌 Concept Shift-Reduce Parsing is a bottom-up parsing technique. It uses a stack and two operations:
• Shift: Push the next input symbol onto the stack.
• Reduce: If the top of the stack matches the RHS of a production, pop those symbols and push the LHS.
Parsing succeeds when only the start symbol remains on the stack and input is exhausted. Productions are entered as E->E+E. 
#include <stdio.h>
#include <string.h>

int i, j, l, n, top = 0, k, p, sl;
char in_str[20], apro[20], pro[20][20], sta[20], stk[20];
int s[10];

void printState() {
    for (i = 0; i <= top; i++) printf("%c", stk[i]);
    printf("\t");
    for (i = l; in_str[i] != '\0'; i++) printf("%c", in_str[i]);
    printf("\t");
}

void shiftReduce() {
    int pp = 0, mm;
    if (in_str[l] != '$') stk[++top] = in_str[l++];
    printf("Shift:%c\n", stk[top]);

    while (1) {
        printState();
        if (stk[top - pp] == '$') { printf("\n"); break; }
        for (mm = top - pp, k = 0; mm <= top; mm++, k++) sta[k] = stk[mm];
        sta[k] = '\0';
        sl = strlen(sta);
        for (i = 0; i < n; i++) {
            if (sl == s[i]) {
                for (j = 3, k = 0; pro[i][j] != '\0'; j++) apro[k++] = pro[i][j];
                apro[k] = '\0';
                if (strcmp(sta, apro) == 0) {
                    top = top - pp; pp = 0;
                    stk[top] = pro[i][0];
                    stk[top+1] = '\0';
                    printf("Reduce: %s\n", pro[i]);
                    printState(); printf("\n");
                }
            }
        }
        pp++;
        if (pp > top + 1) break;
    }
}

int main() {
    printf("Enter number of productions: ");
    scanf("%d", &n);
    printf("Enter productions (e.g. E->E+E):\n");
    for (i = 0; i < n; i++) {
        scanf("%s", pro[i]);
        s[i] = strlen(pro[i]) - 3;
    }
    printf("Enter input string (end with $): ");
    scanf("%s", in_str);

    printf("\nStack\t\tInput\t\tAction\n");
    printf("-----\t\t-----\t\t------\n");

    stk[top] = '$'; l = 0;
    printState(); printf("\n");

    while (in_str[l] != '$') shiftReduce();

    if (stk[top] == pro[0][0] && stk[top+1] == '\0')
        printf("ACCEPT\n");
    else
        printf("ERROR\n");

    return 0;
}
TASK 08 LALR Parser using LEX + YACC
📌 Concept LALR(1) (Look-Ahead LR) is the most commonly used bottom-up parser in practice (used by YACC). It combines LR(0) states with 1-token lookahead. YACC generates the LALR parser automatically from grammar rules. LEX handles tokenization (e.g., recognising numbers), YACC handles the grammar (e.g., expr + term). Compile with: lex file.l && yacc -d file.y && gcc y.tab.c lex.yy.c -o out 
/* ── file: calc.l (LEX file) ── */
%{
#include "y.tab.h"
%}
%%
[0-9]+    { yylval.dval = atof(yytext); return DIGIT; }
\n|.      { return yytext[0]; }
%%
int yywrap() { return 1; }


/* ── file: calc.y (YACC file) ── */
%{
#include <stdio.h>
%}
%union { double dval; }
%token <dval> DIGIT
%type  <dval> expr term factor
%%
line   : expr '\n'          { printf("%g\n", $1); }
       ;
expr   : expr '+' term      { $$ = $1 + $3; }
       | term
       ;
term   : term '*' factor    { $$ = $1 * $3; }
       | factor
       ;
factor : '(' expr ')'      { $$ = $2; }
       | DIGIT
       ;
%%
int main() { yyparse(); return 0; }
void yyerror(char *s) { printf("%s\n", s); }


/* ── Compile commands ── */
// $ lex calc.l
// $ yacc -d calc.y
// $ gcc y.tab.c lex.yy.c -o calc -ll
// $ ./calc
// Input: 4+3   Output: 7
TASK 09 Lexical Analyser using LEX Tool
📌 Concept LEX is a tool that automatically generates a lexical analyser from regular expression-based rules. A .l file has three sections separated by %%: definitions, rules (pattern → action), and user code. The generated yylex() function scans input and fires the matching action. Compile with: lex file.l && gcc lex.yy.c -o out -ll 
/* ── file: lexical.l ── */
%{
#include <stdio.h>
int COMMENT = 0, cnt = 0;
%}

identifier [a-zA-Z][a-zA-Z0-9]*

%%
#.*   { printf("\n%s is a PREPROCESSOR DIRECTIVE", yytext); }

int|float|char|double|while|for|do|if|break|continue|
void|switch|case|long|struct|const|typedef|return|else|goto
      { printf("\n%s is a KEYWORD", yytext); }

"/*"  { COMMENT = 1; }
"*/"  { COMMENT = 0; cnt++; }

{identifier}\(
      { if (!COMMENT) printf("\n%s is a FUNCTION", yytext); }
\{    { if (!COMMENT) printf("\nBLOCK BEGINS"); }
\}    { if (!COMMENT) printf("\nBLOCK ENDS"); }

{identifier}(\[[0-9]*\])?
      { if (!COMMENT) printf("\n%s is an IDENTIFIER", yytext); }

\".*\"
      { if (!COMMENT) printf("\n%s is a STRING", yytext); }

[0-9]+
      { if (!COMMENT) printf("\n%s is a NUMBER", yytext); }

=     { if (!COMMENT) printf("\n%s is an ASSIGNMENT OPERATOR", yytext); }

"<="|">="|"<"|"=="|">"
      { if (!COMMENT) printf("\n%s is a RELATIONAL OPERATOR", yytext); }

\)(\;)?  { if (!COMMENT) { ECHO; printf("\n"); } }
\(    { ECHO; }
[ \t\n]  ;  /* skip whitespace */
%%

int main(int argc, char **argv) {
    if (argc > 1) {
        FILE *fp = fopen(argv[1], "r");

        yyin = fp;

    printf("\n\nTotal comments: %d\n", cnt);
    return 0;
}
int yywrap() { return 1; }

TASK 10 Simple Arithmetic Calculator using LEX
📌 Concept This LEX program implements a calculator by defining patterns for digits and operators. When a digit token is matched, atof() converts the text to float. An operator flag (op) remembers which operation to perform. On newline, the result is printed. This demonstrates how LEX can be used as a simple interpreter without YACC. 
/* ── file: calc.l ── */
%{
#include <stdio.h>
#include <stdlib.h>
#include <math.h>
int op = 0, pw;
float a, b;

void digi() {
    if (op == 0) {
        a = atof(yytext);
    } else {
        b = atof(yytext);
        switch(op) {
            case 1: a = a + b; break;
            case 2: a = a - b; break;
            case 3: a = a * b; break;
            case 4: a = a / b; break;
            case 5: a = pow(a, b); break;
        }
        op = 0;
    }
}
%}

dig  [0-9]+|([0-9]*)"."([0-9]+)

%%
{dig}  { digi(); }
"+"    { op = 1; }
"-"    { op = 2; }
"*"    { op = 3; }
"/"    { op = 4; }
"^"    { op = 5; }
\n     { printf("Result: %f\n\n", a); a = 0; }
[ \t]  ;
%%

int main() { yylex(); return 0; }

TASK 11 Count Characters, Words, Lines, Special Chars using LEX
📌 Concept This LEX program counts statistics of a text file by matching character classes with patterns:
• \n increments line and word counts (end of line = end of a word).
• Tabs/spaces increment word count (word separators).
• Uppercase, lowercase, digits, and special chars are counted separately.
yyin is set to read from a file named myfile.txt. 
/* ── file: count.l ── */
%{
#include <stdio.h>
int lines=0, words=0, upper=0, lower=0, digits=0, special=0;
%}
%%
\n              { lines++; words++; }
[\t ]           { words++; }
[A-Z]           { upper++; }
[a-z]           { lower++; }
[0-9]           { digits++; }
.               { special++; }
%%

int main() {
    yyin = fopen("myfile.txt", "r");
    if (!yyin) { printf("Cannot open myfile.txt\n"); return 1; }
    yylex();
    int total = upper + lower + digits + special;
    printf("This file contains:\n");
    printf("  Lines          : %d\n", lines);
    printf("  Words          : %d\n", words);
    printf("  Uppercase      : %d\n", upper);
    printf("  Lowercase      : %d\n", lower);
    printf("  Digits         : %d\n", digits);
    printf("  Special chars  : %d\n", special);
    printf("  Total chars    : %d\n", total);
    return 0;
}
int yywrap() { return 1; }

/* Compile: lex count.l && gcc lex.yy.c -o count -ll && ./count */
TASK 12 Code Generation for Intermediate Code
📌 Concept Code Generation is the final phase of a compiler. It translates intermediate code (3-address code like a=b+c) into target machine code (assembly-like instructions). Key operations:
• MOV var, R1 → load variable into register.
• ADD/SUB/MUL/DIV src, dst → perform operation.
• MOV R1, var → store result back.
Register allocation is tracked to avoid redundant loads. 
#include <stdio.h>
#include <string.h>

struct { char var; } reg[10];
int noreg = 0;
char st[10][10];
int nost;

char *opcode[4] = {"ADD", "SUB", "MUL", "DIV"};
char  oper[4] = {'+', '-', '*', '/'};

char* toOpcode(char op) {
    for (int i = 0; i < 4; i++)
        if (oper[i] == op) return opcode[i];
    return "???";
}

int inRegister(char var) {
    for (int i = 1; i <= noreg; i++)
        if (reg[i].var == var) return i;
    return 0;
}

int main() {
    int i, regno1 = 1, flag = 0;

    printf("Enter number of 3-address statements: ");
    scanf("%d", &nost);
    printf("Enter statements (e.g. a=b+c):\n");
    for (i = 0; i < nost; i++) scanf("%s", st[i]);

    /* check if consecutive statements share operands (optimization) */
    flag = 0;
    for (i = 0; i < nost - 1; i++)
        if (st[i][2] != st[i+1][2] || st[i][3] != st[i+1][3] || st[i][4] != st[i+1][4])
            flag = 1;

    printf("\n%-15s %s\n", "Statement", "Target Code");
    printf("---------------------------------------------\n");

    for (i = 0; i < nost; i++) {
        printf("%-15s", st[i]);
        /* Load first operand if not already in register */
        if (!inRegister(st[i][2])) {
            noreg++;
            printf("MOV %c, R%d\n", st[i][2], noreg);
            reg[noreg].var = st[i][2];
            regno1 = noreg;
            printf("               ");
        }
        /* Apply operation with second operand */
        if (inRegister(st[i][4])) {
            printf("%s R%d, R%d\n", toOpcode(st[i][3]), inRegister(st[i][4]), regno1);
        } else {
            printf("%s %c, R%d\n", toOpcode(st[i][3]), st[i][4], regno1);
        }
        reg[regno1].var = st[i][0];
        printf("               MOV R%d, %c\n", regno1, st[i][0]);

        if (flag == 0 && i < nost - 1) {
            printf("%-15sMOV R%d, %c\n", st[i+1], regno1, st[i+1][0]);
            break;
        }
    }
    printf("\nCode Generation Completed.\n");
    return 0;
}
Sample input: n=2, statements: a=b+c and x=b+c. Since operands are the same (flag=0), the second result reuses the same register — that's the optimization.
